question - create a responsive gallery page 
1. add multiple image 
2. use flexbox grid
3. apply media queries 

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Responsive Gallery</title>
  <style>
    body {
      font-family: Arial, sans-serif;
      margin: 0;
      padding: 20px;
      background: #f4f4f4;
    }

    .gallery {
      display: flex;
      flex-wrap: wrap; /* allows wrapping to new rows */
      gap: 15px;       /* spacing between items */
      justify-content: center;
    }

    .gallery img {
      width: 100%;
      height: auto;
      border-radius: 8px;
      box-shadow: 0 4px 6px rgba(0,0,0,0.1);
    }

    .gallery-item {
      flex: 1 1 calc(25% - 15px); /* 4 items per row on large screens */
      max-width: calc(25% - 15px);
    }

    /* Tablet view: 2 items per row */
    @media (max-width: 768px) {
      .gallery-item {
        flex: 1 1 calc(50% - 15px);
        max-width: calc(50% - 15px);
      }
    }

    /* Mobile view: 1 item per row */
    @media (max-width: 480px) {
      .gallery-item {
        flex: 1 1 100%;
        max-width: 100%;
      }
    }
  </style>
</head>
